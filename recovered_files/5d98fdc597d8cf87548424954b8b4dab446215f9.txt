const express = require("express");
const router = express.Router();
const passport = require("passport");
const authController = require("../controllers/authController");
const {
  registerValidation,
  loginValidation,
  validate,
} = require("../middlewares/validators");

// Rotas de autenticação
router.get("/login", authController.showLogin);
router.get("/register", authController.showRegister);

router.post("/register", registerValidation, validate, authController.register);

router.post(
  "/login",
  loginValidation,
  validate,
  passport.authenticate("local", {
    successRedirect: "/auth/selecao-perfil",
    failureRedirect: "/auth/login",
    failureFlash: true,
  })
);



// Rota para a página de seleção de perfil
router.get("/selecao-perfil", (req, res) => {
  res.render("auth/selecao-perfil");
});

// Rotas para cadastro específico
router.get("/cadastro-aluno", (req, res) => {
  res.render("auth/register", { role: "aluno" });
});

router.get("/cadastro-professor", (req, res) => {
  res.render("auth/register", { role: "professor" });
});

router.get("/logout", authController.logout);

module.exports = router;
